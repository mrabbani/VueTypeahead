Vuejs and Typeahead Example github source

Default Suggestion

sample code

                <vue-typeahead
                        prefetch="https://twitter.github.io/typeahead.js/data/countries.json"
                        :default-suggestion="true"
                        classes="form-control"
                        v-on:selected="done">

                </vue-typeahead>

Local Values as Default Suggestion

sample code

                <vue-typeahead
                    prefetch="https://twitter.github.io/typeahead.js/data/countries.json"
                    :default-suggestion="true"
                    classes="form-control"
                    :local="localValues"
                    v-on:selected="done">

                </vue-typeahead>

Remote Suggestion

sample code

                <vue-typeahead
                    prefetch="https://twitter.github.io/typeahead.js/data/films/post_1960.json"
                    display-key='value'
                    classes="form-control"
                    remote="https://twitter.github.io/typeahead.js/data/films/queries/%QUERY.json'"
                    v-on:selected="done">

                </vue-typeahead>

Custom template

sample code

                <vue-typeahead
                       v-model="value"
                       prefetch="https://twitter.github.io/typeahead.js/data/nfl.json"
                       :default-suggestion="false"
                       display-key='team'
                       :suggestion-template="myTemplate"
                       classes="form-control"
                       v-on:selected="done">

                </vue-typeahead>
